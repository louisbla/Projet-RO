import java.io.File;
import java.io.FileNotFoundException;
import java.util.Scanner;

/**
 * Created by Marianne
 * on 02/12/2017.
 */
public class Parser {
    private String nameFile;
    private int nbTaches;
    private int tpsTraitement[];
    private int tpsDepart[];
    private int tpsReglage[][];

    public Parser(String nameFile) {
        this.nameFile = nameFile;
        nbTaches = 0;
        Scanner scan;
        File file = new File(nameFile);

        try {
            scan = new Scanner(file);
            nbTaches = scan.nextInt(); // recovering value from first line

            tpsTraitement = new int[nbTaches];
            tpsDepart = new int[nbTaches];
            tpsReglage = new int[nbTaches][nbTaches];

            // recovering value from 2e line (time of treatment)
            for(int secondLine = 0; secondLine < nbTaches; secondLine++){
                tpsTraitement[secondLine] = scan.nextInt();
            }

            // recovering value from 3e line (time of starting)
            for(int thirdLine = 0; thirdLine < nbTaches; thirdLine++){
                tpsDepart[thirdLine] = scan.nextInt();
            }

            // recovering value from the rest of line (time of setting)
            for(int line = 0; line < nbTaches; line++){
                for(int colonn = 0; colonn < nbTaches; colonn++){
                    tpsReglage[line][colonn]= scan.nextInt();
                }
            }



        } catch (FileNotFoundException e1) {
            e1.printStackTrace();
        }
    }

    public int getNbTaches() {
        return nbTaches;
    }

    public int[] getTpsTraitement() {
        return tpsTraitement;
    }

    public int[] getTpsDepart() {
        return tpsDepart;
    }

    public int[][] getTpsReglage() {
        return tpsReglage;
    }
}
